    - DEMIGRAND_AXE,

    - DECOUV_QD, 
    - DES_TEMPO, 
    - MASS_VAL, 
    - MASS_UNIT, 
    - DENSITY, 
    - GRAVITY, 
    - ESCAPE, 
    - VOL_VAL, 
    - VOL_UNIT, 
    - PERIHELION, 
    - APHELION, 
    - ECCENTRICITY, 
    - INCLINATION, 
    - EQUA_RADIUS,
